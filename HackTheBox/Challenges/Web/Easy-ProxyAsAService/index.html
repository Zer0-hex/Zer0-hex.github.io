    发现被过滤了

    • 过滤函数
    RESTRICTED_URLS = ['localhost', '127.', '192.168.', '10.', '172.']
def is_safe_url(url):
    for restricted_url in RESTRICTED_URLS:
        if restricted_url in url:
            return False
    return True

    尝试绕过，发现 8 进制可以
